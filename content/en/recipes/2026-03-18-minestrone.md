  ingredients:
    - "1 can (15 oz) cannellini beans, drained"
    - "1 cup small pasta (ditalini or elbow)"
    - "2 zucchini, diced"
    - "2 carrots, diced"
    - "1 can (14 oz) diced tomatoes"
    - "6 cups vegetable broth"
    - "1 onion, diced"
    - "3 cloves garlic"
    - "1 Parmesan rind (optional)"
    - "2 cups fresh spinach"
    - "Fresh basil and Parmesan for serving"
  steps:
    - "Sauté onion, carrots, and garlic in olive oil. Add zucchini, cook 3 minutes."
    - "Add tomatoes, broth, beans, and Parmesan rind. Bring to a simmer."
    - "Cook 15 minutes until vegetables are tender."
    - "Add pasta and cook until al dente, about 8 minutes."
    - "Stir in spinach. Remove Parmesan rind."
    - "Serve with fresh basil and grated Parmesan."
